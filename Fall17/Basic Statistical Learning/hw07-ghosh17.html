library(caret)
library(mlbench)
library(klaR)

data(Boston, package = "MASS")
set.seed(42)
bstn_idx = createDataPartition(Boston$medv, p = 0.80, list = FALSE)
bstn_trn = Boston[bstn_idx, ]
bstn_tst = Boston[-bstn_idx, ]

# Linear Model
set.seed(1337)
lin_mod = train(
  medv ~ .,
  data = bstn_trn,
  method = "lm",
  trControl = trainControl(method = "cv", number = 5)
)

# Unscaled KNN
set.seed(1337)
bstn_knn_mod = train(
  medv ~ .,
  data = bstn_trn,
  method = "knn",
  trControl = trainControl(method = "cv", number = 5),
  tuneGrid = expand.grid(k = c(1, 5, 10, 15, 20, 25))
)

plot(bstn_knn_mod, xlab="K values", main="Error VS K-Values for un-scaled KNN")

# Scaled KNN
set.seed(1337)
bstn_knn_mod_scaled = train(
  medv ~ .,
  data = bstn_trn,
  method = "knn",
  trControl = trainControl(method = "cv", number = 5),
  preProcess = c("center", "scale"),
  tuneGrid = expand.grid(k = c(1, 5, 10, 15, 20, 25))
)

plot(bstn_knn_mod_scaled, xlab="K values", main="Error VS K-Values for scaled KNN")

#Random Forest
set.seed(1337)
bstn_rf_mod = train(
  medv ~ .,
  data = bstn_trn,
  method = "rf",
  trControl = trainControl(method = "cv", number = 5)
)

#Boosted Tree
gbm_grid = expand.grid(interaction.depth = c(1, 2, 3),
                       n.trees = (1:20) * 100,
                       shrinkage = c(0.1, 0.3),
                       n.minobsinnode = 20)

set.seed(1337)
bstn_gbm_mod = train(
  medv ~ .,
  data = bstn_trn,
  trControl = trainControl(method = "cv", number = 5),
  method = "gbm",
  tuneGrid = gbm_grid, 
  verbose = FALSE
)

plot(bstn_gbm_mod, xlab="n-trees", main="Error VS tuning parameters in Boosted Trees")

calc_rmse = function(actual, predicted) {
  sqrt(mean((actual - predicted) ^ 2))
}

df1 = data.frame(Model_Name = c("linear", "knn-unscaled", "knn-scaled", "random forest", "boosted tree"),
                 Test_Error = c(
                              calc_rmse(actual = bstn_tst$medv, predicted = predict(lin_mod, bstn_tst)),
                              calc_rmse(actual = bstn_tst$medv, predicted = predict(bstn_knn_mod, bstn_tst)),
                              calc_rmse(actual = bstn_tst$medv, predicted = predict(bstn_knn_mod_scaled, bstn_tst)),
                              calc_rmse(actual = bstn_tst$medv, predicted = predict(bstn_rf_mod, bstn_tst)),
                              calc_rmse(actual = bstn_tst$medv, predicted = predict(bstn_gbm_mod, bstn_tst))
                              ),
                 CV_RMSE = c(
                             lin_mod$results$RMSE,
                             min(bstn_knn_mod$results$RMSE),
                             min(bstn_knn_mod_scaled$results$RMSE),
                             min(bstn_rf_mod$results$RMSE),
                             min(bstn_gbm_mod$results$RMSE)
                             )
                 )

set.seed(42)
sim_trn = mlbench::mlbench.2dnormals(n = 750, cl = 5)
sim_trn = data.frame(
  classes = sim_trn$classes,
  sim_trn$x
)

caret::featurePlot(x = sim_trn[, -1], 
                   y = sim_trn$classes, 
                   plot = "pairs",
                   auto.key = list(columns = 2))

set.seed(1337)
sim_lda = train(
  classes ~ .,
  data = sim_trn,
  trControl = trainControl(method = "cv", number = 10),
  method = "lda",
  verbose = FALSE
)

set.seed(1337)
sim_qda = train(
  classes ~ .,
  data = sim_trn,
  trControl = trainControl(method = "cv", number = 10),
  method = "qda",
  verbose = FALSE
)

set.seed(1337)
sim_nb = train(
  classes ~ .,
  data = sim_trn,
  trControl = trainControl(method = "cv", number = 10),
  method = "nb",
  verbose = FALSE
)

set.seed(1337)
sim_rda = train(
  classes ~ .,
  data = sim_trn,
  trControl = trainControl(method = "cv", number = 10),
  method = "rda",
  verbose = FALSE
)

plot(sim_rda, main = "Accuracy vs Tuning parameters plot")

df2 = data.frame(Model_Name = c("LDA", "QDA", "Naive Bayes", "RDA"),
                 CV_Accuracy = c(sim_lda$results$Accuracy,

                                 max(sim_rda$results$Accuracy)),
                 Accuracy_Standard_Deviation = c(
